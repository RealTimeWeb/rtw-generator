001    package realtimeweb.weatherservice.domain;
002    
003    import java.util.ArrayList;
004    import java.util.Arrays;
005    import java.util.HashMap;
006    import java.util.List;
007    import java.util.Map;
008    
009    
010    
011    
012    /**
013     * A prediction for future weather.
014     */
015    public class Forecast {
016            
017        private String longDescription;
018        private String description;
019        private String imageUrl;
020        private String temperatureLabel;
021        private String periodName;
022        // This value can sometimes be null, so you should set that to 0.
023        private Integer probabilityOfPrecipitation;
024        private String periodTime;
025        private Integer temperature;
026        
027        
028        /*
029         * @return A more-detailed, human-readable description of the predicted weather for this period.
030         */
031        public String getLongDescription() {
032            return this.longDescription;
033        }
034        
035        /*
036         * @param A more-detailed, human-readable description of the predicted weather for this period.
037         * @return String
038         */
039        public void setLongDescription(String longDescription) {
040            this.longDescription = longDescription;
041        }
042        
043        /*
044         * @return A human-readable description of the predicted weather for this period.
045         */
046        public String getDescription() {
047            return this.description;
048        }
049        
050        /*
051         * @param A human-readable description of the predicted weather for this period.
052         * @return String
053         */
054        public void setDescription(String description) {
055            this.description = description;
056        }
057        
058        /*
059         * @return A url pointing to a picture that describes the predicted weather for this period.
060         */
061        public String getImageUrl() {
062            return this.imageUrl;
063        }
064        
065        /*
066         * @param A url pointing to a picture that describes the predicted weather for this period.
067         * @return String
068         */
069        public void setImageUrl(String imageUrl) {
070            this.imageUrl = imageUrl;
071        }
072        
073        /*
074         * @return Either 'High' or 'Low', depending on whether or not the predicted temperature is a daily high or a daily low.
075         */
076        public String getTemperatureLabel() {
077            return this.temperatureLabel;
078        }
079        
080        /*
081         * @param Either 'High' or 'Low', depending on whether or not the predicted temperature is a daily high or a daily low.
082         * @return String
083         */
084        public void setTemperatureLabel(String temperatureLabel) {
085            this.temperatureLabel = temperatureLabel;
086        }
087        
088        /*
089         * @return A human-readable name for this time period (e.g. Tonight or Saturday).
090         */
091        public String getPeriodName() {
092            return this.periodName;
093        }
094        
095        /*
096         * @param A human-readable name for this time period (e.g. Tonight or Saturday).
097         * @return String
098         */
099        public void setPeriodName(String periodName) {
100            this.periodName = periodName;
101        }
102        
103        /*
104         * @return The probability of precipitation for this period (as a percentage).
105         */
106        public Integer getProbabilityOfPrecipitation() {
107            return this.probabilityOfPrecipitation;
108        }
109        
110        /*
111         * @param The probability of precipitation for this period (as a percentage).
112         * @return Integer
113         */
114        public void setProbabilityOfPrecipitation(Integer probabilityOfPrecipitation) {
115            this.probabilityOfPrecipitation = probabilityOfPrecipitation;
116        }
117        
118        /*
119         * @return A string representing the time that this period starts. Encoded as YYYY-MM-DDTHH:MM:SS, where the T is not a number, but a always present character (e.g. 2013-07-30T18:00:00).
120         */
121        public String getPeriodTime() {
122            return this.periodTime;
123        }
124        
125        /*
126         * @param A string representing the time that this period starts. Encoded as YYYY-MM-DDTHH:MM:SS, where the T is not a number, but a always present character (e.g. 2013-07-30T18:00:00).
127         * @return String
128         */
129        public void setPeriodTime(String periodTime) {
130            this.periodTime = periodTime;
131        }
132        
133        /*
134         * @return The predicted temperature for this period (in Fahrenheit).
135         */
136        public Integer getTemperature() {
137            return this.temperature;
138        }
139        
140        /*
141         * @param The predicted temperature for this period (in Fahrenheit).
142         * @return Integer
143         */
144        public void setTemperature(Integer temperature) {
145            this.temperature = temperature;
146        }
147        
148            
149            /**
150             * Creates a string based representation of this Forecast.
151            
152             * @return String
153             */
154            public String toString() {
155                    return "Report[" +longDescription+", "+description+", "+imageUrl+", "+temperatureLabel+", "+periodName+", "+probabilityOfPrecipitation+", "+periodTime+", "+temperature+"]";
156            }
157            
158            /**
159             * Internal constructor to create a Forecast from a json representation.
160             * @param map The raw json data that will be parsed.
161             * @return 
162             */
163        
164        public Forecast(Map<String, Object> raw) {
165            
166            this.longDescription = ((Map<String, Object>) raw.get("data")).get("text").toString();
167            this.description = ((Map<String, Object>) raw.get("data")).get("weather").toString();
168            this.imageUrl = ((Map<String, Object>) raw.get("data")).get("iconLink").toString();
169            this.temperatureLabel = ((Map<String, Object>) raw.get("time")).get("tempLabel").toString();
170            this.periodName = ((Map<String, Object>) raw.get("time")).get("startPeriodName").toString();
171            this.probabilityOfPrecipitation = Integer.parseInt(((Map<String, Object>) raw.get("data")).get("pop").toString());
172            this.periodTime = ((Map<String, Object>) raw.get("time")).get("startValidTime").toString();
173            this.temperature = Integer.parseInt(((Map<String, Object>) raw.get("data")).get("temperature").toString());
174        
175            }       
176    }
